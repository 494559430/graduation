create table emp(
    empid int PRIMARY key auto_increment,
    empname VARCHAR(32),
    idcard varchar(32),
    sex int,#0代表女1代表男
    age int,
    email varchar(32),
    loc VARCHAR(200),
    phone varchar(32),
    hiredate varchar(32),
    sal varchar(32),
    leavedate varchar(32),
    jobId VARCHAR(32),
    state int #1代表离职2为普通店员3为店长
);
create table job(
    jobId int PRIMARY key auto_increment,
    jobName VARCHAR(32),
    shopid int
);
CREATE table supplier(
    supplierId int PRIMARY key auto_increment,
    supplierName VARCHAR(32),
    representative VARCHAR(32),
    supplierEmail VARCHAR(32),
    supplierPhone varchar(32)
);

CREATE TABLE `shopitem`  (
  `shopItemid` int(11) NULL DEFAULT NULL,
  `shopid` int(11) NULL DEFAULT NULL,
  `num` int(11) NULL DEFAULT NULL,
  `ifgrounding` int(11) NULL DEFAULT NULL,
  `by1` varchar(100) CHARACTER SET utf8 COLLATE utf8_general_ci NULL DEFAULT NULL,
  `by2` varchar(100) CHARACTER SET utf8 COLLATE utf8_general_ci NULL DEFAULT NULL
) ENGINE = InnoDB CHARACTER SET = utf8 COLLATE = utf8_general_ci ROW_FORMAT = Compact;

CREATE TABLE `shopitemdescrip`  (
  `shopItemid` int(11) NOT NULL AUTO_INCREMENT,
  `shopItemname` varchar(32) CHARACTER SET utf8 COLLATE utf8_general_ci NULL DEFAULT NULL,
  `price` double NULL DEFAULT NULL,
  `supplierId` int(11) NULL DEFAULT NULL,
  `shopItemType` int(11) NULL DEFAULT NULL,
  `descrip` varchar(1000) CHARACTER SET utf8 COLLATE utf8_general_ci NULL DEFAULT NULL,
  `by1` varchar(100) CHARACTER SET utf8 COLLATE utf8_general_ci NULL DEFAULT NULL,
  `by2` varchar(100) CHARACTER SET utf8 COLLATE utf8_general_ci NULL DEFAULT NULL,
  PRIMARY KEY (`shopItemid`) USING BTREE
) ENGINE = InnoDB AUTO_INCREMENT = 12 CHARACTER SET = utf8 COLLATE = utf8_general_ci ROW_FORMAT = Compact;

create table shop(
